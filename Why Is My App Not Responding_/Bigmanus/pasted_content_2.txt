🔧 Technical Implementation
Reverse Proxy Configuration
# nginx.conf for lectureme.org
server {
    listen 80;
    server_name lectureme.org www.lectureme.org;
    
    # Your custom frontend
    location / {
        root /var/www/lectureme-frontend;
        try_files $uri $uri/ /index.html;
    }
    
    # Proxy API calls to Manus
    location /api/ {
        proxy_pass https://your-app.manus.space/api/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }
}
